Demo Anime.js với Hiệu Ứng Từng Chữ Cái

C h à o m ừ n g đ ến v ới h i ệ u ứng t ừng c h ữ !

Mã nguồn dưới đây sẽ được làm nổi bật:


        // Đây là mã JavaScript để hiển thị từng chữ cái khi lướt đến
        const text = document.getElementById('animated-text');
        const letters = text.querySelectorAll('.letter');

        const animateLetters = () => {
            letters.forEach((letter, index) => {
                anime({
                    targets: letter,
                    opacity: [0, 1], // Từ ẩn sang hiện
                    translateY: [20, 0], // Di chuyển từ dưới lên
                    duration: 500,
                    delay: index * 200, // Độ trễ cho từng chữ cái
                    easing: 'easeInOutExpo'
                });
            });
        };

        window.addEventListener('scroll', () => {
            const rect = text.getBoundingClientRect();
            if (rect.top < window.innerHeight && rect.bottom >= 0) {
                animateLetters(); // Gọi hàm khi cuộn đến văn bản
                window.removeEventListener('scroll', arguments.callee); // Ngăn chặn việc gọi lại
            }
        });
